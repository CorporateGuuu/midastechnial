<?php
function nexustechhub_usa_theme_setup() {
    // Add theme support
    add_theme_support('woocommerce');
    add_theme_support('title-tag');
    add_theme_support('post-thumbnails');
    
    // Enqueue styles
    wp_enqueue_style('main-style', get_stylesheet_uri(), array(), '1.0');
    
    // Enqueue custom JS (optional)
    wp_enqueue_script('custom-js', get_template_directory_uri() . '/js/custom.js', array('jquery'), '1.0', true);
    
    // Register navigation menu
    register_nav_menus(array(
        'primary' => __('Primary Menu', 'nexustechhub-usa-theme'),
    ));
}
add_action('wp_enqueue_scripts', 'nexustechhub_usa_theme_setup');

// Add WooCommerce support for product gallery
add_theme_support('wc-product-gallery-zoom');
add_theme_support('wc-product-gallery-lightbox');
add_theme_support('wc-product-gallery-slider');
?>
